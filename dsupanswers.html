from abc import ABC, abstractmethod
print('vnvr')
class Appliance(ABC):
    def __init__(self, name):
        self.name = name

    @abstractmethod
    def turn_on(self):
        pass

class WashingMachine(Appliance):
    def turn_on(self):
        print(f"Turning on {self.name}")
class Refrigirator(Appliance):
    def turn_on(self):
        print(f"Turning on {self.name}")

washing_machine = WashingMachine("Washing Machine", 110)
washing_machine.turn_on()

refrigerator = Refrigirator("Refrigerator", 220)
refrigerator.turn_on()

word = input('Enter a Word: ')
letter = input('Enter a letter u want to search: ')
def number_of_occurrences(letter,word):
    count = 0
    for i in word:
        if i == letter:
            count += 1
    return count

letter_count = number_of_occurrences(letter, word)

print(f'The letter {letter} appears {letter_count} times in the {word}.')

class Calculator:
def __init__(self,a,b):
    self.a = a
    self.b = b
def add(self,a,b):
    return self.a + self.b
def subtract(self,a,b):
    return self.a - self.b
def multiply(self,a,b):
    return self.a * self.b
def divide(self,a,b):
    if b == 0:
        return 'Error: Division by zero is not allowed.'
    else:
        return self.a / self.b
a = int(input('Enter a number'))
b = int(input('Enter a number'))
calculator = Calculator(a, b)
print(calculator.add(a,b))
print(calculator.subtract(a,b))
print(calculator.multiply(a,b))
print(calculator.divide())
